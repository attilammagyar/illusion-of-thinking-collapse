def towers_of_hanoi(n, source=0, destination=2, auxiliary=1):
    """
    Solve the Tower of Hanoi puzzle for n disks.
    
    Args:
        n: Number of disks
        source: Source peg (0-indexed)
        destination: Destination peg (0-indexed)
        auxiliary: Auxiliary peg (0-indexed)
    
    Returns:
        List of moves in the format [disk, from_peg, to_peg]
    """
    moves = []
    
    def _hanoi(n, source, destination, auxiliary):
        if n == 0:
            return
        
        _hanoi(n-1, source, auxiliary, destination)
        moves.append([n, source, destination])
        _hanoi(n-1, auxiliary, destination, source)
    
    _hanoi(n, source, destination, auxiliary)
    return moves
