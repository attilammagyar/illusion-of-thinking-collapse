function generateHanoiMoves(n, source, auxiliary, destination) {
    const moves = [];
    
    function hanoiHelper(disks, src, aux, dst) {
        if (disks === 0) return;
        
        hanoiHelper(disks - 1, src, dst, aux);
        moves.push(`${disks},${src},${dst}`);
        hanoiHelper(disks - 1, aux, src, dst);
    }
    
    hanoiHelper(n, source, auxiliary, destination);
    return moves;
}
